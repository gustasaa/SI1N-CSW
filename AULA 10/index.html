  def celsius_para_fahrenheit(celsius):
    """
    Converte uma temperatura de graus Celsius para Fahrenheit.
    
    Parâmetros:
    celsius (float): Temperatura em graus Celsius.
    
    Retorna:
    float: Temperatura em graus Fahrenheit.
    """
    fahrenheit = (celsius * 9/5) + 32
    return fahrenheit

# Exemplo de uso:
temperatura_celsius = float(input("Digite a temperatura em graus Celsius: "))
temperatura_fahrenheit = celsius_para_fahrenheit(temperatura_celsius)
print(f"A temperatura em graus Fahrenheit é: {temperatura_fahrenheit:.2f}")
